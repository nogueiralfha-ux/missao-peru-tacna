const CACHE_NAME = 'missao-peru-v2';
const ASSETS = [
  'index.html',
  'devocionais.json',
  'questions.json',
  'logo-missao-peru.png',
  'manifest.json'
];

// Instala o Service Worker e armazena os arquivos essenciais no cache
self.addEventListener('install', (e) => {
  e.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      console.log('Missão Peru: Arquivos armazenados para modo offline');
      return cache.addAll(ASSETS);
    })
  );
});

// Remove caches antigos quando houver atualizações
self.addEventListener('activate', (e) => {
  e.waitUntil(
    caches.keys().then((keys) => {
      return Promise.all(
        keys.filter((key) => key !== CACHE_NAME).map((key) => caches.delete(key))
      );
    })
  );
});

// Tenta carregar do cache primeiro, se falhar busca na rede
self.addEventListener('fetch', (e) => {
  e.respondWith(
    caches.match(e.request).then((response) => {
      return response || fetch(e.request);
    })
  );
});
